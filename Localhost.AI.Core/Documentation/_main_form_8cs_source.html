83 private async Task ActionLauncherAsync()
84 {
85 // Check if there's text in the input box
86 if (!string.IsNullOrWhiteSpace(txtBoxAsk.Text))
87 {
88 // Handle special commands
89 if (txtBoxAsk.Text.ToLower().Trim() == "/help")
90 ShowControl(_helpControl);
91 else if (txtBoxAsk.Text.ToLower().Trim() == "/entities")
92 ShowControl(_EntitiesControl);
93 else if (txtBoxAsk.Text.ToLower().Trim() == "/symbolic")
94 ShowControl(_symbolicProcessor);
95 else
96 {
97 // Create a request object for the AI model
98 Request request = new Request
99 {
100 model = "mistral-small3.1",
101 messages = new List<Localhost.AI.Core.Models.LLM.Message>
102 {
103 new Localhost.AI.Core.Models.LLM.Message
104 {
105 role = "user",
106 content = txtBoxAsk.Text.Trim()
107 }
108 }
109 };
110
111 // Call the Ollama API to get a response
112 Response response = await OllamaManager.CallOllamaAsync(_config.DialogServerUrl, request);
113
114 // Check if we got a valid response
115 if (response != null && response.choices != null && response.choices.Count > 0)
116 {
117 // Build the response text from all choices
118 StringBuilder sb = new StringBuilder();
119 foreach (var choice in response.choices)
120 {
121 sb.AppendLine(choice.message.content);
122 }
123 labelAnswer.Text = sb.ToString();
124
125 // Create a robot request for text-to-speech
126 RequestRobot req = new RequestRobot
127 {
128 action = "voice",
129 text = labelAnswer.Text,
130 parameters = "",
131 language = "fr"
132 };
133
134 // Send the text to the robot for voice synthesis
135 RestHelper.PostAsync("http://localhost:8080/api/process", req);
136 }
137 else
138 {
139 // Display error message if no response received
140 labelAnswer.Text = "No response from model.";
141 }
142 }
143 }
144 else
145 {
146 // Do nothing if input is empty
147 }
148 }
